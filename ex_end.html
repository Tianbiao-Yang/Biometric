#import data
mydataframe <- read.csv("q2_data.csv")
#split data in groups
exc <- mydataframe$Performance_Scores[mydataframe$Treatment=="excitingDrug"]
anx <- mydataframe$Performance_Scores[mydataframe$Treatment=="anxietyDrug"]
hyper <- mydataframe$Performance_Scores[mydataframe$Treatment=="hypertensionDrug"]
pla <- mydataframe$Performance_Scores[mydataframe$Treatment=="Placebos"]
#make a group list
drugs <- list(exc, anx, hyper, pla)
#calculate group means, variances and number of samples
mean_value <- sapply(drugs, function(x)mean(x))
var_value <- sapply(drugs, function(x)var(x))
length_value <- sapply(drugs, function(x)length(x))
#calculate whole mean
grandmean <- mean(mean_value)
#calculate df
df_wt <- length(mydataframe$Performance_Scores)-length(mean_value)
df_bt <- length(mean_value)-1
#calculate SS
SStotal <- sum((mydataframe$Performance_Scores - grandmean)^2)
SSwt <- sum((length_value-1)*var_value)
SSbt <- SStotal - SSwt
#calculate MS
MSbt <- SSbt/df_bt
MSwt <- SSwt/df_wt
#Calculate F
F_value <- MSbt/MSwt
#Calculate P-value
1-pf(F_value, df1 = df_bt, df2 = df_wt)

t_test <- function(x) {
  x <- unlist(x)
  if (var.test(x[1:161],x[162:346])$p.value < 0.05) {
    t.test(x[1:161],x[162:346],var.equal = F,
           alternative = "less")$p.value
  }else {
    t.test(x[1:161],x[162:346],var.equal = T,
           alternative = "less")$p.value
  }
}
